Modal with images

Click on any image below

Code:

var images = document.getElementsByTagName('img');
    var images = document.getElementsByTagName('img');
    for(var i = 0; i<images.length; i++) {
        images[i].onclick = function() {
           new MessageBox({
                header: this.getAttribute('alt'),
                body: "<img src='" + this.src + "' width='300' />",
                loadAJAX: false,
                draggable: true
            });
        }
    }
